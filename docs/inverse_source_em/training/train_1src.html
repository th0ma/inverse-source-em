inverse_source_em.training.train_1src

Training script for Regression Problem I (single-source localization).

This script trains the MultitaskNet model to jointly predict: - x, y : Cartesian coordinates of the source - I : source strength (intensity)

The training objective combines three components:

  1. Localization loss (x, y): L_xy = ||(x_pred, y_pred) - (x_true, y_true)||^2

  2. Strength loss (I): L_I = ||I_pred - I_true||^2

  3. Field-consistency loss:

    • Convert (x_pred, y_pred) → (rho_pred, phi_pred)
    • Use surrogate forward model to compute predicted fields
    • Compare to true normalized fields in X_fields

All three losses are combined using heteroscedastic uncertainty weighting (Kendall & Gal, 2018), with learnable log-variance parameters: s_xy, s_I, s_f

The script performs: - dataset loading - train/validation split - surrogate loading - multitask loss computation - Adam optimization - ReduceLROnPlateau scheduling - early stopping - checkpoint saving

103def polar_from_cartesian(x, y):
104    """
105    Convert Cartesian coordinates (x, y) to polar coordinates (rho, phi).
106
107    Parameters
108    ----------
109    x : torch.Tensor of shape (B,)
110        Predicted x-coordinates.
111    y : torch.Tensor of shape (B,)
112        Predicted y-coordinates.
113
114    Returns
115    -------
116    rho : torch.Tensor of shape (B,)
117        Radial distance sqrt(x^2 + y^2).
118    phi : torch.Tensor of shape (B,)
119        Angle atan2(y, x) in radians.
120
121    Notes
122    -----
123    - This conversion is used to feed predictions into the surrogate
124      forward model for field-consistency loss.
125    """
126    rho = torch.sqrt(x**2 + y**2)
127    phi = torch.atan2(y, x)
128    return rho, phi

131def multitask_loss(model_output, X_fields, Y_true, theta_obs, ymax_E, ymax_H):
132    """
133    Compute the multitask loss for Regression I.
134
135    The loss consists of three components:
136
137    1. Localization loss (x, y):
138           L_xy = ||(x_pred, y_pred) - (x_true, y_true)||^2
139
140    2. Strength loss (I):
141           L_I = ||I_pred - I_true||^2
142
143    3. Field-consistency loss:
144           - Convert (x_pred, y_pred) → (rho_pred, phi_pred)
145           - Use surrogate forward model to compute predicted fields
146           - Compare to true normalized fields in X_fields
147
148    All three losses are combined using heteroscedastic uncertainty
149    weighting with learnable log-variances (s_xy, s_I, s_f):
150
151        L_total =
152            0.5 * exp(-s_xy) * L_xy + 0.5 * s_xy +
153            0.5 * exp(-s_I)  * L_I  + 0.5 * s_I  +
154            0.5 * exp(-s_f)  * L_fields + 0.5 * s_f
155
156    Parameters
157    ----------
158    model_output : dict
159        Output of MultitaskNet:
160            {
161                "xy": (B, 2),
162                "I":  (B, 1),
163                "s_xy": scalar,
164                "s_I":  scalar,
165                "s_f":  scalar,
166                "h":    (B, hidden_dim)
167            }
168    X_fields : torch.Tensor of shape (B, 4*M)
169        True boundary fields:
170            [E_r, E_i, H_r, H_i] concatenated.
171    Y_true : torch.Tensor of shape (B, 3)
172        True regression targets: [x, y, I].
173    theta_obs : torch.Tensor of shape (M,)
174        Observation angles used by the surrogate.
175    ymax_E, ymax_H : float
176        Normalization constants for E and H fields.
177
178    Returns
179    -------
180    L_total : torch.Tensor (scalar)
181        Total multitask loss.
182    metrics : dict
183        Dictionary of scalar floats for logging:
184            "L_total", "L_xy", "L_I", "L_fields",
185            "s_xy", "s_I", "s_f"
186
187    Notes
188    -----
189    - The surrogate forward model is accessed via the global `sur` object.
190    - All computations use float64 for numerical stability.
191    """
192    xy_pred = model_output["xy"]
193    I_pred  = model_output["I"].view(-1)
194
195    s_xy = model_output["s_xy"]
196    s_I  = model_output["s_I"]
197    s_f  = model_output["s_f"]
198
199    # True targets
200    x_true = Y_true[:, 0]
201    y_true = Y_true[:, 1]
202    I_true = Y_true[:, 2]
203
204    # Localization loss
205    x_pred = xy_pred[:, 0]
206    y_pred = xy_pred[:, 1]
207    L_xy = ((x_pred - x_true)**2 + (y_pred - y_true)**2).mean()
208
209    # Strength loss
210    L_I = ((I_pred - I_true)**2).mean()
211
212    # Field-consistency loss
213    M = theta_obs.shape[0]
214
215    E_r_true = X_fields[:, 0:M]
216    E_i_true = X_fields[:, M:2*M]
217    H_r_true = X_fields[:, 2*M:3*M]
218    H_i_true = X_fields[:, 3*M:4*M]
219
220    rho_pred, phi_pred = polar_from_cartesian(x_pred, y_pred)
221
222    E_r_pred, E_i_pred, H_r_pred, H_i_pred = sur.batch_forward(
223        rho_pred, phi_pred, I_pred, theta_obs
224    )
225
226    E_r_pred /= ymax_E
227    E_i_pred /= ymax_E
228    H_r_pred /= ymax_H
229    H_i_pred /= ymax_H
230
231    L_fields = (
232        (E_r_pred - E_r_true)**2 +
233        (E_i_pred - E_i_true)**2 +
234        (H_r_pred - H_r_true)**2 +
235        (H_i_pred - H_i_true)**2
236    ).mean()
237
238    # Uncertainty-weighted total loss
239    L_total = (
240        0.5 * torch.exp(-s_xy) * L_xy + 0.5 * s_xy +
241        0.5 * torch.exp(-s_I)  * L_I  + 0.5 * s_I  +
242        0.5 * torch.exp(-s_f)  * L_fields + 0.5 * s_f
243    )
244
245    return L_total, {
246        "L_total":  L_total.item(),
247        "L_xy":     L_xy.item(),
248        "L_I":      L_I.item(),
249        "L_fields": L_fields.item(),
250        "s_xy":     s_xy.item(),
251        "s_I":      s_I.item(),
252        "s_f":      s_f.item(),
253    }
254

285def main():
286    """
287    Main training routine for Regression I.
288
289    Steps
290    -----
291    1. Load dataset_1src.npz via Regression1SrcDataset.
292    2. Split into training and validation subsets.
293    3. Load surrogate forward model (Esurf, Hsurf).
294    4. Instantiate MultitaskNet.
295    5. Train for up to 200 epochs with:
296           - Adam optimizer
297           - ReduceLROnPlateau scheduler
298           - Early stopping (patience = 20)
299    6. Save best-performing checkpoints.
300    7. Print training progress and validation loss.
301
302    Notes
303    -----
304    - The surrogate is used inside the loss function for field consistency.
305    - All tensors are cast to float64 for consistency with surrogate models.
306    - Checkpoints are saved in models/regression_1src/.
307    """
308    global best_val_loss, epochs_no_improve
309
310    for epoch in range(1, 201):
311
312        # TRAIN
313        model.train()
314        train_losses = []
315
316        for X_batch, Y_batch in train_loader:
317            X_batch = X_batch.to(device, dtype=torch.float64)
318            Y_batch = Y_batch.to(device, dtype=torch.float64)
319
320            optimizer.zero_grad()
321
322            out = model(X_batch)
323            loss, _ = multitask_loss(out, X_batch, Y_batch, theta_obs, ymax_E, ymax_H)
324
325            loss.backward()
326            optimizer.step()
327
328            train_losses.append(loss.item())
329
330        train_loss = np.mean(train_losses)
331
332        # VAL
333        model.eval()
334        val_losses = []
335
336        with torch.no_grad():
337            for X_batch, Y_batch in val_loader:
338                X_batch = X_batch.to(device, dtype=torch.float64)
339                Y_batch = Y_batch.to(device, dtype=torch.float64)
340
341                out = model(X_batch)
342                loss, _ = multitask_loss(out, X_batch, Y_batch, theta_obs, ymax_E, ymax_H)
343                val_losses.append(loss.item())
344
345        val_loss = np.mean(val_losses)
346        scheduler.step(val_loss)
347
348        # Checkpoint
349        if val_loss < best_val_loss - 1e-5:
350            best_val_loss = val_loss
351            epochs_no_improve = 0
352
353            torch.save(
354                model.state_dict(),
355                os.path.join(CHECKPOINT_DIR, f"best_epoch_{epoch:04d}.pth")
356            )
357
358            print(f"[epoch {epoch:04d}] train={train_loss:.6f} val={val_loss:.6f}")
359
360        else:
361            epochs_no_improve += 1
362
363        if epochs_no_improve >= patience:
364            print("Early stopping.")
365            break
366
367    print("Training completed.")
368
369
370if __name__ == "__main__":
371    main()
